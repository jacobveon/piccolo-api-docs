piccolo-pressure-detect.lua

--[[
Performs a pressure detect at different locations in a zone.

Logs a warning when the pressure threshold is not exceeded.
]]

detectPressureZSpeed = 1.0
detectPressureZAcceleration = 200.0
maxZDepth = 20
threshold = 10
tipZone = "Zone1"

--[[
Slows the z axis movement and performs a pressure detect.

Returns success/failure.
]]
function detectPressure(zone, position, threshold, maxZDepth)
    flow.go(zone, position)

    -- slow down the z axis
    veon.spdacc("z", detectPressureZSpeed, detectPressureZAcceleration)

    flow.log(0, "Pressure above ".. zone .. ":" .. position ..": " .. piccolo.getpressure())

    local res = piccolo.pressuredetect(zone, flow.resolveAddress(zone, position), maxZDepth, 0, threshold)

    if res == 0 then
        flow.log(1, "Failed to detect pressure above " .. threshold .. " at " .. zone .. ":" .. position)
    end

    flow.log(0, "Pressure at " .. zone .. ":" .. position .. ": " .. piccolo.getpressure())

    -- reset to normal speed
    flow.setMasterSpeed()

    return res
end

function main()
    flow.getTip(tipZone, "", true)

    detectPressure("Zone2", "A1", maxZDepth, threshold)
    detectPressure("Zone2", "A8", maxZDepth, threshold)
    detectPressure("Zone2", "A6", maxZDepth, threshold)

    flow.log(0, "Ejecting tip")

    flow.ejectTip()
end

flow.initialise()

main()
