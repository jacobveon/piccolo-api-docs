piccolo-mix.lua

--[[
How to perform a mix using the piccolo module.

The flow module offers a helper (@see flow.mix) function to perform mixing.
]]

function mix(volume, cycles, mixTipDepth, flowRate, tracked)
    veon.sendrawpiccolocommand(7001, volume*1000, 4)
    veon.sendrawpiccolocommand(7003, flowRate*1000, 4)
    veon.sendrawpiccolocommand(7007, flowRate*1000, 4)

    for i = 1, cycles do
        veon.sendrawpiccolocommand(10)
        veon.sendrawpiccolocommand(11)
    end
end

local volume = 5
local aspirateInsertionDepth = 10

local tipZone = "Zone1"

local source = {
    zone = "Zone2",
    position = "A1",
    insertionDepth = 20
}

local destination = {
    zone = "Zone3",
    position = "A1"
}

flow.initialise()

flow.log(0, "Picking up tip from " .. tipZone)

-- seek for a tip
flow.getTip(tipZone, "", true)

piccolo.configureaspirate(volume)

flow.go(source.zone, source.position)

flow.log(0, "Aspirating " .. volume .. "uL from " .. source.zone .. ":" .. source.position)

veon.gorel(0, 0, source.insertionDepth)

veon.getvol(volume, 0, false)

-- reverse back out of the liquid
veon.gorel(0, 0, (source.insertionDepth*-1))

piccolo.starttransportliquid()

flow.go(destination.zone, destination.position)

piccolo.stoptransportliquid()

flow.log(0, "Dispensing " .. volume .. "uL in " .. destination.zone .. ":" .. destination.position)

piccolo.dispensenoncontact()

veon.gorel(0, 0, 8)

mix(50, 3, 0, 200, 0)

veon.gorel(0, 0, -8)

flow.log(0, "Ejecting tip")

flow.ejectTip()
