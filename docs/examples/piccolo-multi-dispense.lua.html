piccolo-multi-dispense.lua

--[[
Performs a multi-dispense transfer.
]]
local volume = 30
local dispenseVolume = 10
local aspirateInsertionDepth = 10

local tipZone = "Zone1"

local source = {
    zone = "Zone2",
    position = "A1",
    insertionDepth = 20
}

local destinations = {
    {
        zone = "Zone3",
        position = "A1"
    },
    {
        zone = "Zone3",
        position = "A2"
    },
    {
        zone = "Zone3",
        position = "A3"
    }
}

flow.initialise()

flow.log(0, "Picking up tip from " .. tipZone)

-- seek for a tip
flow.getTip(tipZone, "", true)

piccolo.configureaspirate(volume)

flow.go(source.zone, source.position)

flow.log(0, "Aspirating " .. volume .. "uL from " .. source.zone .. ":" .. source.position)

veon.gorel(0, 0, source.insertionDepth)

veon.getvol(volume, 0, false)

-- reverse back out of the liquid
veon.gorel(0, 0, (source.insertionDepth*-1))

for i, destination in ipairs(destinations) do
    piccolo.starttransportliquid()

    flow.go(destination.zone, destination.position)

    piccolo.stoptransportliquid()

    flow.log(0, "Dispensing " .. dispenseVolume .. "uL in " .. destination.zone .. ":" .. destination.position)

    piccolo.dispensenoncontactmultistage(dispenseVolume)
end

flow.log(0, "Ejecting tip")

flow.ejectTip()
